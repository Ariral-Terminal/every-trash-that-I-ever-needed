### Базовый шаблон:

```
"""
    📝 ModuleName - краткое описание
    
    Подробное описание функционала модуля
    для Heroku UserBot
    
    ⚠️ Предупреждения (если нужны)
"""

__version__ = (1, 0, 0)

# meta developer: @your_username
# meta pic: https://link-to-icon.png (опционально)
# meta banner: https://link-to-banner.png (опционально)
# scope: hikka_only (опционально)
# scope: hikka_min 1.2.0 (минимальная версия, опционально)
# requires: package1 package2 (зависимости PyPI)

from .. import loader, utils
from herokutl.types import Message
import logging

logger = logging.getLogger(__name__)

@loader.tds
class ModuleNameMod(loader.Module):
    """Краткое описание модуля для списка модулей"""
    
    strings = {
        "name": "ModuleName",
        "key1": "Текст на английском",
        "key2": "Другой текст",
    }
    
    strings_ru = {
        "name": "ModuleName",
        "key1": "Текст на русском",
        "key2": "Другой текст",
    }

    def __init__(self):
        self.config = loader.ModuleConfig(
            loader.ConfigValue(
                "parameter_name",
                default_value,
                "Описание параметра",
                validator=loader.validators.Type()
            )
        )

    async def client_ready(self, client, db):
        """Инициализация при запуске"""
        self._client = client
        self._db = db

    @loader.command(
        ru_doc="Описание команды на русском",
        en_doc="Command description in English"
    )
    async def commandcmd(self, message: Message):
        """Command description"""
        # Логика команды
        await utils.answer(message, self.strings("key1"))
```

## ОБЯЗАТЕЛЬНЫЕ ЭЛЕМЕНТЫ

### 1. Декораторы и метаданные:
- `@loader.tds` - делает модуль translateable (обязательно)
- `__version__ = (major, minor, patch)` - версия модуля
- `# meta developer:` - указание автора
- `# requires:` - зависимости через пробел

### 2. Класс модуля:
```
@loader.tds
class YourModuleMod(loader.Module):
    """Краткое описание для списка"""
```

### 3. Локализация (strings):
```
strings = {
    "name": "ModuleName",
    "message_key": "Default English text",
}

strings_ru = {
    "message_key": "Русский текст",
}

strings_uz = {
    "message_key": "O'zbek matn",
}
```

### 4. Команды:
```
@loader.command(
    ru_doc="<аргументы> - описание",
    en_doc="<args> - description"
)
async def commandnamecmd(self, message: Message):
    """English description"""
    args = utils.get_args_raw(message)
    
    if not args:
        await utils.answer(message, self.strings("error_no_args"))
        return
    
    try:
        # Основная логика
        result = await some_function(args)
        await utils.answer(message, f"✅ {result}")
    except Exception as e:
        logger.exception(e)
        await utils.answer(message, f"❌ {self.strings('error')}: {e}")
```

## РАБОТА С БАЗОЙ ДАННЫХ

### Прямой доступ:
```
# Чтение
value = self._db.get("ModuleName", "key", default_value)

# Запись
self._db.set("ModuleName", "key", value)

# Pointer (ссылка на значение)
pointer = self._db.pointer("ModuleName", "key", default_value)
```

### Через обёртки (рекомендуется):
```
# Чтение
value = self.get("key", default_value)

# Запись
self.set("key", value)

# Pointer для изменяемых типов (list, dict)
my_list = self.pointer("users", [])
my_list.append(user_id)  # Автоматически сохраняется
```

### Пример работы с pointer:
```
async def client_ready(self, client, db):
    self._client = client
    self._db = db
    
    # Создаём pointer на список
    self.users = self.pointer("users_list", [])

async def addusercmd(self, message: Message):
    """Добавить пользователя"""
    args = utils.get_args_raw(message)
    
    if args not in self.users:
        self.users.append(args)  # Автоматом сохраняется в БД
        await utils.answer(message, f"✅ Добавлен: {args}")
    else:
        await utils.answer(message, "⚠️ Уже есть")
```

## ЧАСТЫЕ ОШИБКИ

❌ **НЕ ДЕЛАЙ ТАК:**
```
# 1. Забыл @loader.tds
class MyMod(loader.Module):  # Ошибка!

# 2. Неправильное имя команды
async def command(self, message):  # Должно быть commandcmd

# 3. Забыл async
def commandcmd(self, message):  # Должно быть async def

# 4. Не проверил аргументы
args = utils.get_args_raw(message)
result = some_function(args)  # Может быть пусто!

# 5. Забыл обработать ошибки
response = await api_call()  # Может упасть!
```

✅ **ДЕЛАЙ ТАК:**
```
# 1. С декоратором
@loader.tds
class MyMod(loader.Module):

# 2. Правильное имя
async def commandcmd(self, message: Message):

# 3. С async
async def commandcmd(self, message: Message):

# 4. С проверкой
args = utils.get_args_raw(message)
if not args:
    await utils.answer(message, "Укажи аргументы!")
    return

# 5. С try-except
try:
    response = await api_call()
except Exception as e:
    logger.exception(e)
    await utils.answer(message, f"Ошибка: {e}")
    return
```
